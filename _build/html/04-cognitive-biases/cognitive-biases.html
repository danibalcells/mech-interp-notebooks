# Function to get a response from the model
def get_response(prompt):
    response = client.chat.completions.create(
        model="gpt-3.5-turbo",  # Use the appropriate model
        messages=[
            {"role": "system", "content": "Answer the following question in a few words with the first answer you can think of."},
            {"role": "user", "content": prompt}
        ],
        max_tokens=150
    )
    return response.choices[0].message.content

# Iterate through scenarios and get responses for both frames
for i, scenario in enumerate(scenarios):
    print(scenario)
    gain_response = get_response(scenario["gain_frame"])
    loss_response = get_response(scenario["loss_frame"])
    
    print(f"Scenario {i+1}:")
    print(f"Gain Frame Response: {gain_response}")
    print(f"Loss Frame Response: {loss_response}")
    print()
